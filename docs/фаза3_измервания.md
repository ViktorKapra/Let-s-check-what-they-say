## Помощни функции за измерване

```python
# src/benchmark_utils.py
"""
Помощни функции за всички измервания.
Единствен source of truth за timing логиката.
"""
import time
import statistics
import json
from pathlib import Path
from typing import Callable, Any


def measure(fn: Callable, *args, repeats: int = 3, **kwargs) -> dict:
    """
    Изпълнява функцията `repeats` пъти и връща статистики.

    Аргументи:
        fn:      функцията за измерване
        *args:   позиционни аргументи към fn
        repeats: брой повторения (препоръчително ≥ 3)
        **kwargs: именовани аргументи към fn

    Връща речник с:
        times:   списък с всички времена
        mean:    средно аритметично
        stdev:   стандартно отклонение
        min:     минимално
        max:     максимално
        result:  резултатът от последното изпълнение
    """
    times = []
    result = None

    for i in range(repeats):
        start = time.perf_counter()
        result = fn(*args, **kwargs)
        elapsed = time.perf_counter() - start
        times.append(elapsed)
        print(f"    Повторение {i+1}/{repeats}: {elapsed:.3f} сек.")

    return {
        "times":  times,
        "mean":   statistics.mean(times),
        "stdev":  statistics.stdev(times) if len(times) > 1 else 0.0,
        "min":    min(times),
        "max":    max(times),
        "result": result,
    }


def speedup(t_serial: float, t_parallel: float) -> float:
    """Speedup(p) = T_serial / T_parallel(p)"""
    return t_serial / t_parallel if t_parallel > 0 else float("inf")


def efficiency(spdup: float, n_workers: int) -> float:
    """Efficiency(p) = Speedup(p) / p"""
    return spdup / n_workers if n_workers > 0 else 0.0


def save_results(results: dict, path: Path) -> None:
    """Записва речник с резултати като JSON."""
    path.parent.mkdir(parents=True, exist_ok=True)
    # Премахваме несериализируеми стойности (DataFrame, numpy масиви)
    clean = {}
    for k, v in results.items():
        try:
            json.dumps(v)
            clean[k] = v
        except (TypeError, ValueError):
            clean[k] = str(type(v).__name__)
    with open(path, "w", encoding="utf-8") as f:
        json.dump(clean, f, ensure_ascii=False, indent=2)
    print(f"  Резултати записани в: {path}")
```

```python
"""
Е1: Измерване на speedup при асинхронно теглене.

Сравнява:
  - Серийно теглене (requests, последователно)
  - asyncio с различна конкурентност (1, 5, 10, 20 едновременни заявки)

Забележка: Това измерване зависи от мрежата и сървъра на strazha.bg.
Изпълни го при стабилна мрежа и запиши резултатите веднага.

        "stdev": serial_stats["stdev"],

    for i, concurrency in enumerate([1, 5, 10, 20], start=2):
        print(f"\n[{i}/5] asyncio (конкурентност={concurrency})...")
        stats = measure(
            lambda: asyncio.run(scrape_session_async(TEST_DATE, RAW_DIR, max_concurrent=concurrency)),
            repeats=REPEATS,
        )
        sp   = speedup(t_serial, stats["mean"])
        eff  = efficiency(sp, concurrency)
        all_results[f"async_{concurrency}"] = {
            "concurrency": concurrency,
            "mean":        stats["mean"],
            "stdev":       stats["stdev"],
            "times":       stats["times"],
            "speedup":     sp,
            "efficiency":  eff,
        }
        print(f"  Средно: {stats['mean']:.2f} ± {stats['stdev']:.2f} сек. "
              f"| Speedup: {sp:.2f}x | Efficiency: {eff:.2f}")

    save_results(all_results, RESULTS_DIR / "e1_io_benchmark.json")
    return all_results
```

## Експеримент 5 — Amdahl анализ (серийна фракция)

Amdahl's Law казва, че максималният speedup е ограничен от серийната фракция на кода. Този експеримент измерва колко от общото времетраене е серийно (агрегирането) и колко е паралелизируемо (токенизацията).

```python
# scripts/benchmark_amdahl.py
"""
Е5: Amdahl анализ — измерване на серийната фракция.

Декомпозира времетраенето на run_serial() на три стъпки:
  A: Токенизация (паралелизируема с multiprocessing)
  Б: Агрегиране по партия (серийно — изисква синхронизация)
  В: TF-IDF изчисление (паралелизируемо с vectorization)

Серийна фракция f = T_serial_steps / T_total
Теоретичен максимален speedup = 1 / f  (закон на Amdahl)
"""
import sys, time
sys.path.insert(0, "src")

import pandas as pd
from pathlib import Path
from collections import Counter
from common import load_speeches, tokenize, make_bigrams, make_trigrams
from analysis_serial import aggregate_by_party, compute_tfidf
from benchmark_utils import save_results

CORPORA_DIR = Path("data/corpora")
RESULTS_DIR = Path("results")
REPEATS     = 3


def measure_steps(df: pd.DataFrame) -> dict:
    """Измерва времетраенето на всяка отделна стъпка."""
    speeches = df.to_dict("records")

    # Стъпка A: токенизация (паралелизируема)
    t_a_list = []
    for _ in range(REPEATS):
        start = time.perf_counter()
        speech_results = [
            {
                "party":    s["party"],
                "unigrams": Counter(tokenize(s["text"])),
                "bigrams":  Counter(make_bigrams(tokenize(s["text"]))),
                "trigrams": Counter(make_trigrams(tokenize(s["text"]))),
                "total_words": len(tokenize(s["text"])),
            }
            for s in speeches
        ]
        t_a_list.append(time.perf_counter() - start)

    # Стъпка Б: агрегиране (серийно)
    t_b_list = []
    for _ in range(REPEATS):
        start = time.perf_counter()
        by_party = aggregate_by_party(speech_results)
        t_b_list.append(time.perf_counter() - start)

    # Стъпка В: TF-IDF (паралелизируема с vectorization)
    t_c_list = []
    for _ in range(REPEATS):
        start = time.perf_counter()
        compute_tfidf(by_party)
        t_c_list.append(time.perf_counter() - start)

    import statistics
    t_a = statistics.mean(t_a_list)
    t_b = statistics.mean(t_b_list)
    t_c = statistics.mean(t_c_list)
    t_total = t_a + t_b + t_c

    # Серийна фракция = само Стъпка Б
    f = t_b / t_total

    # Теоретичен максимален speedup по Amdahl
    import math
    max_speedup = 1 / f if f > 0 else float("inf")

    return {
        "n_speeches":  len(speeches),
        "t_A_tokenize_sec":   round(t_a, 4),
        "t_B_aggregate_sec":  round(t_b, 4),
        "t_C_tfidf_sec":      round(t_c, 4),
        "t_total_sec":        round(t_total, 4),
        "pct_A": round(100 * t_a / t_total, 1),
        "pct_B": round(100 * t_b / t_total, 1),
        "pct_C": round(100 * t_c / t_total, 1),
        "serial_fraction_f":       round(f, 4),
        "amdahl_max_speedup":      round(max_speedup, 2),
    }


def run_amdahl_benchmarks():
    print("=" * 60)
    print("Е5 — Amdahl анализ (серийна фракция)")
    print("=" * 60)

    all_results = {}

    for corpus_name in ["small", "medium", "large"]:
        path = CORPORA_DIR / f"{corpus_name}.csv"
        if not path.exists():
            continue

        df = pd.read_csv(path, encoding="utf-8")
        print(f"\nКорпус: {corpus_name} ({len(df)} изказвания)")

        r = measure_steps(df)
        all_results[corpus_name] = r

        print(f"  Стъпка A (токенизация):    {r['t_A_tokenize_sec']:.4f} сек. "
              f"({r['pct_A']:.1f}%) ← паралелизируема")
        print(f"  Стъпка Б (агрегиране):     {r['t_B_aggregate_sec']:.4f} сек. "
              f"({r['pct_B']:.1f}%) ← серийна")
        print(f"  Стъпка В (TF-IDF):         {r['t_C_tfidf_sec']:.4f} сек. "
              f"({r['pct_C']:.1f}%) ← паралелизируема")
        print(f"  ─────────────────────────────────────")
        print(f"  Серийна фракция f:         {r['serial_fraction_f']:.4f}")
        print(f"  Макс. теорет. speedup:     {r['amdahl_max_speedup']:.2f}x")

    save_results(all_results, RESULTS_DIR / "e5_amdahl_benchmark.json")
    return all_results


if __name__ == "__main__":
    run_amdahl_benchmarks()
```

## Очаквани резултати и интерпретация

| Експеримент | Очакван резултат | Защо |
|-------------|-----------------|------|
| Е1 asyncio (малък) | 5–10x speedup | I/O-bound, latency скрита зад конкурентност |
| Е2 Pool (малък) | < 1x или ~1x | Overhead > полза при ~280 изказвания |
| Е2 Pool (голям) | 2–4x speedup | Полезен при много данни |
| Е3 Vectorized | 10–50x speedup | C код vs Python цикли |
| Е4 Dask (малък) | < 1x | Overhead на партицирането > полза |
| Е4 Dask (голям) | 1.5–3x | Полезен при наистина голям корпус |
| Е5 Amdahl | f ≈ 0.05–0.15 | Агрегирането е малка фракция → висок теорет. макс. speedup |

Ако резултатите се различават значително от очакваните — **това е интересен материал за доклада**, не грешка. Обясни защо.
